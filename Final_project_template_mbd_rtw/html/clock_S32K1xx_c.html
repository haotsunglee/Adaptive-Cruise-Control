601status_t CLOCK_DRV_Init(clock_manager_user_config_t const * config)

615

634/*FUNCTION**********************************************************************



635 *



636 * Function Name : CLOCK_SYS_SetScgConfiguration



637 * Description   : This function configures the SCG blocks



638 *



639 * END**************************************************************************/



640static status_t CLOCK_SYS_SetScgConfiguration(const scg_config_t * scgConfig)



641{



642    status_t status = STATUS_SUCCESS;



643    DEV_ASSERT(scgConfig != NULL);



644



645    if (scgConfig != NULL)



646    {



647        /* Configure a temporary system clock source: FIRC */



648        status = CLOCK_SYS_ConfigureTemporarySystemClock();



649



650        if (status == STATUS_SUCCESS)



651        {



652            /* Configure clock sources from SCG */



653            status = CLOCK_SYS_ConfigureModulesFromScg(scgConfig);



654        }



655



656        if (status == STATUS_SUCCESS)



657        {



658            g_RtcClkInFreq = 0U; /* reset the value RTC_clk frequency. */



659            if (scgConfig->rtcConfig.initialize)



660            {



661                /* RTC Clock settings. */



662                g_RtcClkInFreq = scgConfig->rtcConfig.rtcClkInFreq;



663            }



664



665            /* Configure SCG ClockOut. */



666            if (scgConfig->clockOutConfig.initialize)



667            {



668                /* ClockOut settings. */



669                SCG_SetClockoutSourceSel(SCG, (uint32_t)scgConfig->clockOutConfig.source);



670            }



671



672            /* Configure SCG clock modes. */



673            if (scgConfig->clockModeConfig.initialize)



674            {



675                /* Configure SCG clock modes */



676                status = CLOCK_SYS_SetSystemClockConfig(SCG_SYSTEM_CLOCK_MODE_RUN, &(scgConfig->clockModeConfig.rccrConfig));



677                if (status == STATUS_SUCCESS)



678                {



679                    status = CLOCK_SYS_SetSystemClockConfig(SCG_SYSTEM_CLOCK_MODE_VLPR, &(scgConfig->clockModeConfig.vccrConfig));



680                }



681



682#if FEATURE_HAS_HIGH_SPEED_RUN_MODE



683                if (status == STATUS_SUCCESS)



684                {



685                    status = CLOCK_SYS_SetSystemClockConfig(SCG_SYSTEM_CLOCK_MODE_HSRUN, &(scgConfig->clockModeConfig.hccrConfig));



686                }



687#endif



688            }



689        }



690    }



691



692    return status;



693}

784        SIM_SetTraceClockSource(SIM, simClockConfig->traceClockConfig.source);

791                                    simClockConfig->traceClockConfig.divEnable,



792                                    simClockConfig->traceClockConfig.divider,



793                                    simClockConfig->traceClockConfig.divFraction);



794        }



795    }



796}

2845                                               scg_system_clock_config_t const * config)



2846{



2847    status_t status = STATUS_SUCCESS;



2848    uint32_t srcFreq = 0U;



2849



2850    /* The maximum clock frequencies of system clocks in all power modes */



2851    static const uint32_t maxSysClksInVLPR[MODES_MAX_NO][SYS_CLK_MAX_NO]  = CLOCK_MAX_FREQUENCIES_VLPR_MODE;



2852    static const uint32_t maxSysClksInRUN[MODES_MAX_NO][SYS_CLK_MAX_NO]   = CLOCK_MAX_FREQUENCIES_RUN_MODE;

2854    static const uint32_t maxSysClksInHSRUN[MODES_MAX_NO][SYS_CLK_MAX_NO] = CLOCK_MAX_FREQUENCIES_HSRUN_MODE;



2855#endif



2856    const uint32_t sysFreqMul = ((uint32_t)config->divCore) + 1UL;



2857    const uint32_t busFreqMul = (((uint32_t)config->divCore) + 1UL) * (((uint32_t)config->divBus) + 1UL);



2858    const uint32_t slowFreqMul = (((uint32_t)config->divCore) + 1UL) * (((uint32_t)config->divSlow) + 1UL);

2870                (srcFreq > (busFreqMul  * (maxSysClksInRUN[(uint32_t)config->src][BUS_CLK_INDEX] >> 4U)))  ||    /* Bus clock */

2882            DEV_ASSERT(SCG_SYSTEM_CLOCK_SRC_SIRC    == config->src);



2883            /* Verify the frequencies of sys, bus and slow clocks. */



2884            if ((srcFreq > (sysFreqMul  * (maxSysClksInVLPR[(uint32_t)config->src][CORE_CLK_INDEX] >> 4U))) ||    /* Sys(core) clock */



2885                (srcFreq > (busFreqMul  * (maxSysClksInVLPR[(uint32_t)config->src][BUS_CLK_INDEX] >> 4U)))  ||    /* Bus clock */



2886                (srcFreq > (slowFreqMul * (maxSysClksInVLPR[(uint32_t)config->src][SLOW_CLK_INDEX] >> 4U))))      /* Slow clock */



2887            {



2888                /* Configuration for the next system clock source is not valid. */



2889                status = STATUS_ERROR;



2890            }



2891            else



2892            {



2893                SCG_SetVlprClockControl(SCG, (uint32_t)config->src, (uint32_t)config->divCore, (uint32_t)config->divBus, (uint32_t)config->divSlow);



2894            }



2895            break;



2896#if FEATURE_HAS_HIGH_SPEED_RUN_MODE



2897        case SCG_SYSTEM_CLOCK_MODE_HSRUN:     /*!< High Speed Run mode.     */



2898            DEV_ASSERT((SCG_SYSTEM_CLOCK_SRC_FIRC == config->src) || ( SCG_SYSTEM_CLOCK_SRC_SYS_PLL == config->src));



2899            /* Verify the frequencies of sys, bus and slow clocks. */



2900            if ((srcFreq > (sysFreqMul  * (maxSysClksInHSRUN[(uint32_t)config->src][CORE_CLK_INDEX] >> 4U))) ||    /* Sys(core) clock */



2901                (srcFreq > (busFreqMul * (maxSysClksInHSRUN[(uint32_t)config->src][BUS_CLK_INDEX] >> 4U)))  ||    /* Bus clock */



2902                (srcFreq > (slowFreqMul * (maxSysClksInHSRUN[(uint32_t)config->src][SLOW_CLK_INDEX] >> 4U))))      /* Slow clock */

2904                /* Configuration for the next system clock source is not valid. */

2906            }

2908            {

2910            }



2911            break;



2912#endif



2913        default:

2917    }



2918    return status;



2919}
