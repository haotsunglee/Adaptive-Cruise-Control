32/*FUNCTION**********************************************************************



33 *



34 * Function Name : BIT



35 * Description   : Return bit B in byte A



36 * This is not a public API as it is called by other API functions.



37 *



38 *END**************************************************************************/



39static inline uint8_t BIT(uint8_t A,



40                          uint8_t B)



41{



42    return (uint8_t)((A >> B) & 0x01U);



43}



44



45/*FUNCTION**********************************************************************



46 *



47 * Function Name : LIN_DRV_ProcessParity



48 * Description   : Makes or checks parity bits. If action is checking parity, the function



49 * returns ID value if parity bits are correct or 0xFF if parity bits are incorrect. If action



50 * is making parity bits, then from input value of ID, the function returns PID.



51 * This is not a public API as it is called by other API functions.



52 *



53 * Implements    : LIN_DRV_ProcessParity_Activity



54 *END**************************************************************************/



55uint8_t LIN_DRV_ProcessParity(uint8_t PID,



56                              uint8_t typeAction)



57{



58    uint8_t parity;



59    uint8_t retVal;



60



61    parity = (uint8_t)(((0xFFU & (BIT(PID, 0U) ^ BIT(PID, 1U) ^ BIT(PID, 2U) ^ BIT(PID, 4U))) << 6U) |



62                       ((0xFFU ^ (BIT(PID, 1U) ^ BIT(PID, 3U) ^ BIT(PID, 4U) ^ BIT(PID, 5U))) << 7U));



63



64    /* Check if action is checking parity bits */



65    if (CHECK_PARITY == typeAction)



66    {



67        /* If parity bits are incorrect */



68        if ((PID & 0xC0U) != parity)



69        {



70            /* Return 0xFF if parity bits are incorrect */



71            retVal = 0xFFU;



72        }



73        /* If parity bits are correct */



74        else



75        {



76            /* Return ID if parity bits are correct */



77            retVal = (uint8_t)(PID & 0x3FU);



78        }



79    }



80    /* If action is making parity bits */



81    else



82    {



83        /* Return PID in case of making parity bits */



84        retVal = (uint8_t)(PID | parity);



85    }



86



87    return retVal;



88}

90/*FUNCTION**********************************************************************



91 *



92 * Function Name : LIN_DRV_MakeChecksumByte



93 * Description   : Makes the checksum byte for a frame. For PID of identifiers,



94 * if PID is 0x3C (ID 0x3C) or 0x7D (ID 0x3D) or 0xFE (ID 0x3E) or 0xBF (ID 0x3F)



95 * apply classic checksum and apply enhanced checksum for other PID.



96 * In case user want to calculate classic checksum please set PID to zero.



97 *



98 * Implements    : LIN_DRV_MakeChecksumByte_Activity



99 *END**************************************************************************/



100uint8_t LIN_DRV_MakeChecksumByte(const uint8_t * buffer,



101                                 uint8_t sizeBuffer,



102                                 uint8_t PID)



103{



104    uint8_t length;



105    uint16_t checksum = 0U;



106



107    /* For PID is 0x3C (ID 0x3C) or 0x7D (ID 0x3D) or 0xFE (ID 0x3E) or 0xBF (ID 0x3F)



108     * apply classic checksum and apply enhanced checksum for other PID */



109    if ((0x3CU != PID) && (0x7DU != PID) && (0xFEU != PID) && (0xBFU != PID))



110    {



111        /* For PID other than 0x3C, 0x7D, 0xFE and 0xBF: Add PID in checksum calculation */



112        checksum = PID;



113    }



114    else



115    {



116        /* For 0x3C, 0x7D, 0xFE and 0xBF: Do not add PID in checksum calculation */



117        checksum = 0U;



118    }



119



120    for (length = sizeBuffer; 0U < length; length--)



121    {



122        checksum += *buffer;



123        buffer++;



124        /* Deal with the carry */



125        if (checksum > 0xFFU)



126        {



127            checksum -= 0xFFU;



128        }



129    }



130



131    /* Return reversed checksum */



132    return (uint8_t)(~checksum);



133}
