89/*!



90 * @brief Halts or does not halt the eDMA module when an error occurs.



91 *



92 * An error causes the HALT bit to be set. Subsequently, all service requests are ignored until the



93 * HALT bit is cleared.



94 *



95 * @param base Register base address for eDMA module.



96 * @param haltOnError Halts (true) or not halt (false) eDMA module when an error occurs.



97 */



98static inline void EDMA_SetHaltOnErrorCmd(DMA_Type * base, bool haltOnError)



99{



100    uint32_t regValTemp;



101#ifdef FEATURE_DMA_HWV3



102    regValTemp = base->MP_CSR;



103    regValTemp &= ~(DMA_MP_CSR_HAE_MASK);



104    regValTemp |= DMA_MP_CSR_HAE(haltOnError ? 1UL : 0UL);



105    base->MP_CSR = regValTemp;



106#else



107    regValTemp = base->CR;



108    regValTemp &= ~(DMA_CR_HOE_MASK);



109    regValTemp |= DMA_CR_HOE(haltOnError ? 1UL : 0UL);



110    base->CR = regValTemp;



111#endif



112}

121/*!



122 * @brief Sets the eDMA channel priority.



123 *



124 * @param base Register base address for eDMA module.



125 * @param channel eDMA channel number.



126 * @param priority Priority of the DMA channel. Different channels should have different priority



127 * setting inside a group.



128 */



129static inline void EDMA_SetChannelPriority(DMA_Type * base, uint8_t channel, edma_channel_priority_t priority)



130{



131#if defined (CUSTOM_DEVASSERT) || defined (DEV_ERROR_DETECT)



132    DEV_ASSERT(channel < FEATURE_DMA_CHANNELS);



133#endif



134



135#ifdef FEATURE_DMA_HWV3



136    volatile uint32_t regValTemp;



137    regValTemp = base->TCD[channel].CH_PRI;



138    regValTemp &= (uint32_t)~(DMA_TCD_CH_PRI_APL_MASK);



139    regValTemp |= (uint32_t)DMA_TCD_CH_PRI_APL(priority);



140    base->TCD[channel].CH_PRI = regValTemp;



141#else



142    uint8_t regValTemp;



143    uint8_t index = (uint8_t)FEATURE_DMA_CHN_TO_DCHPRI_INDEX(channel);



144    regValTemp = base->DCHPRI[index];



145    regValTemp &= (uint8_t)~(DMA_DCHPRI_CHPRI_MASK);



146    regValTemp |= (uint8_t)DMA_DCHPRI_CHPRI(priority);



147    base->DCHPRI[index] = regValTemp;



148#endif



149}

169    base->CR = regValTemp;



170#endif

174/*!



175 * @brief Gets the channel arbitration algorithm.



176 *



177 * @param base Register base address for eDMA module.



178 * @return edma_arbitration_algorithm_t variable indicating the selected



179 * channel arbitration: Round-Robin way or fixed priority way



180 */



181static inline edma_arbitration_algorithm_t EDMA_GetChannelArbitrationMode(const DMA_Type * base)



182{



183    edma_arbitration_algorithm_t retVal;



184#ifdef FEATURE_DMA_HWV3



185    if (((base->MP_CSR >> DMA_MP_CSR_ERCA_SHIFT) & 1U) != 0U)



186#else



187    if (((base->CR >> DMA_CR_ERCA_SHIFT) & 1U) != 0U)



188#endif



189    {



190        retVal = EDMA_ARBITRATION_ROUND_ROBIN;



191    }



192    else



193    {



194        retVal = EDMA_ARBITRATION_FIXED_PRIORITY;



195    }



196    return retVal;



197}

259    regValTemp |= DMA_TCD_NBYTES_MLOFFNO_SMLOE(enable ? 1UL : 0UL);



260    regValTemp |= DMA_TCD_NBYTES_MLOFFNO_DMLOE(enable ? 1UL : 0UL);
