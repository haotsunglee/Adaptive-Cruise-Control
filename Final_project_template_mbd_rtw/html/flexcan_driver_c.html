303 *



304 * Function Name : FLEXCAN_DRV_SetRxFifoGlobalMask



305 * Description   : Set Rx FIFO global mask as the 11-bit standard mask or the



306 * 29-bit extended mask.



307 *



308 * Implements    : FLEXCAN_DRV_SetRxFifoGlobalMask_Activity



309 *END**************************************************************************/



310void FLEXCAN_DRV_SetRxFifoGlobalMask(



311    uint8_t instance,



312    flexcan_msgbuff_id_type_t id_type,



313    uint32_t mask)



314{



315    DEV_ASSERT(instance < CAN_INSTANCE_COUNT);



316    flexcan_rx_fifo_id_element_format_t formatType;



317    CAN_Type * base = g_flexcanBase[instance];



318    uint32_t calcMask = 0U;



319



320    FLEXCAN_EnterFreezeMode(base);



321



322    if (true == FLEXCAN_IsRxFifoEnabled(base))



323    {



324		formatType = FLEXCAN_GetRxFifoIdFormat(base);



325		calcMask = FLEXCAN_GetRxFifoMask(id_type, formatType, mask);



326



327		switch (formatType)



328		{



329			case FLEXCAN_RX_FIFO_ID_FORMAT_A :



330				FLEXCAN_SetRxFifoGlobalMask(base, calcMask);



331				break;



332			case FLEXCAN_RX_FIFO_ID_FORMAT_B :



333				FLEXCAN_SetRxFifoGlobalMask(base, (calcMask | (calcMask >> FLEXCAN_RX_FIFO_ID_FILTER_FORMATB_EXT_SHIFT1)));



334				break;



335			case FLEXCAN_RX_FIFO_ID_FORMAT_C :



336				FLEXCAN_SetRxFifoGlobalMask(base, (calcMask | (calcMask >> FLEXCAN_RX_FIFO_ID_FILTER_FORMATC_SHIFT1) |



337												  	  	  	  (calcMask >> FLEXCAN_RX_FIFO_ID_FILTER_FORMATC_SHIFT2) |



338															  (calcMask >> FLEXCAN_RX_FIFO_ID_FILTER_FORMATC_SHIFT3)));



339				break;



340			default :



341				/* Will Enable all filter fields mask */



342				FLEXCAN_SetRxFifoGlobalMask(base, 0xFFFFFFFFU);



343				break;



344		}



345    }



346    FLEXCAN_ExitFreezeMode(base);



347}

627/*FUNCTION**********************************************************************

629 * Function Name : FLEXCAN_DRV_Init



630 * Description   : Initialize FlexCAN driver.



631 * This function will select a source clock, reset FlexCAN module, set maximum



632 * number of message buffers, initialize all message buffers as inactive, enable



633 * RX FIFO if needed, mask all mask bits, disable all MB interrupts, enable



634 * FlexCAN normal mode, and enable all the error interrupts if needed.



635 *



636 * Implements    : FLEXCAN_DRV_Init_Activity

640   flexcan_state_t *state,

644    DEV_ASSERT(state != NULL);



645    DEV_ASSERT(g_flexcanStatePtr[instance] == NULL);



646#ifdef ERRATA_E10595



647    if (instance != 0U)



648    {



649    	DEV_ASSERT((MC_ME->GS & MC_ME_GS_S_FXOSC_MASK) != 0U);



650    }



651#endif



652



653    status_t result;



654    CAN_Type * base = g_flexcanBase[instance];



655    flexcan_time_segment_t bitrate;



656    status_t osifStat;



657    uint32_t i, j;



658



659    if(FLEXCAN_IsEnabled(base))



660    {



661        /* To enter Disable Mode requires FreezMode first */



662        FLEXCAN_EnterFreezeMode(base);



663        FLEXCAN_Disable(base);

2543        tmpBitrate = clkFreq / ((tmpPresdiv + 1U) * numTq);



2544



2545        /* The number of time quanta in 1 bit time must be lower than the one supported */



2546        if ((numTq >= FLEXCAN_NUM_TQ_MIN) && (numTq < FLEXCAN_NUM_TQ_MAX))



2547        {



2548            /* Compute time segments based on the value of the sampling point */



2549            tSeg1 = (numTq * samplePoint / 100U) - 1U;



2550            tSeg2 = numTq - 1U - tSeg1;



2551



2552            /* Adjust time segment 1 and time segment 2 */



2553            FLEXCAN_ProccessTSeg(&tSeg1, &tSeg2);



2554



2555            tmpPseg2 = tSeg2 - 1U;



2556



2557            /* Start from pseg1 = pseg2 and adjust until propseg is valid */



2558            tmpPseg1 = tmpPseg2;



2559            tmpPropseg = tSeg1 - tmpPseg1 - 2U;



2560



2561            FLEXCAN_ProccessPSeg(&tmpPropseg, &tmpPseg1);



2562            



2563



2564            if (((tSeg1 >= FLEXCAN_TSEG1_MAX) || (tSeg2 >= FLEXCAN_TSEG2_MAX) || (tSeg2 < FLEXCAN_TSEG2_MIN) || (tSeg1 < FLEXCAN_TSEG1_MIN)) ||



2565               ((tmpPropseg >= FLEXCAN_PROPSEG_MAX) || (tmpPseg1 >= FLEXCAN_PSEG1_MAX) || (tmpPseg2 < FLEXCAN_PSEG2_MIN) || (tmpPseg2 >= FLEXCAN_PSEG2_MAX)))



2566            {



2567                continue;



2568            }



2569



2570            tmpSample = ((tSeg1 + 1U) * 100U) / numTq;



2571            dSample = FLEXCAN_CheckDsample(tmpSample , samplePoint);



2572            dBitrate = FLEXCAN_CheckdBitrate(tmpBitrate , bitrate);



2573



2574            if ((dBitrate < dBitrateMin) ||



2575                ((dBitrate == dBitrateMin) && (dSample < dSampleMin)))



2576            {



2577                dSampleMin = dSample;



2578                dBitrateMin = dBitrate;



2579                pseg1 = tmpPseg1;



2580                pseg2 = tmpPseg2;



2581                presdiv = tmpPresdiv;



2582                propseg = tmpPropseg;



2583



2584                if ((dBitrate == 0U) && (dSample <= 1U))



2585                {



2586                    break;



2587                }



2588            }



2589        }



2590    }



2591



2592    timeSeg->phaseSeg1 = pseg1;



2593    timeSeg->phaseSeg2 = pseg2;



2594    timeSeg->preDivider = presdiv;



2595    timeSeg->propSeg = propseg;



2596    timeSeg->rJumpwidth = FLEXCAN_CheckJumpwidth(pseg1);



2597    return tmpBitrate;



2598}
