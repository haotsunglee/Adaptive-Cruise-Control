71/*FUNCTION**********************************************************************



72 *



73 * Function Name : init_data_bss



74 * Description   : Make necessary initializations for RAM.



75 * - Copy the vector table from ROM to RAM.



76 * - Copy initialized data from ROM to RAM.



77 * - Copy code that should reside in RAM from ROM



78 * - Clear the zero-initialized data section.



79 *



80 * Tool Chains:



81 *   __GNUC__           : GNU Compiler Collection



82 *   __ghs__            : Green Hills ARM Compiler



83 *   __ICCARM__         : IAR ARM Compiler



84 *   __DCC__            : Wind River Diab Compiler



85 *   __ARMCC_VERSION    : ARMC Compiler



86 *



87 * Implements    : init_data_bss_Activity



88 *END**************************************************************************/



89void init_data_bss(void)



90{



91    uint32_t n;



92    uint8_t coreId;



93/* For ARMC we are using the library method of initializing DATA, Custom Section and



94 * Code RAM sections so the below variables are not needed */



95#if !defined(__ARMCC_VERSION)



96    /* Declare pointers for various data sections. These pointers



97     * are initialized using values pulled in from the linker file */



98    uint8_t * data_ram;



99    uint8_t * code_ram;



100    uint8_t * bss_start;



101    uint8_t * custom_ram;



102    const uint8_t * data_rom, * data_rom_end;



103    const uint8_t * code_rom, * code_rom_end;



104    const uint8_t * bss_end;



105    const uint8_t * custom_rom, * custom_rom_end;



106#endif



107    /* Addresses for VECTOR_TABLE and VECTOR_RAM come from the linker file */



108



109#if defined(__ARMCC_VERSION)



110    extern uint32_t __RAM_VECTOR_TABLE_SIZE;



111    extern uint32_t __VECTOR_ROM;



112    extern uint32_t __VECTOR_RAM;



113#else



114    extern uint32_t __RAM_VECTOR_TABLE_SIZE[];



115    extern uint32_t __VECTOR_TABLE[];



116    extern uint32_t __VECTOR_RAM[];



117#endif



118    /* Get section information from linker files */



119#if defined(__ICCARM__)



120    /* Data */



121    data_ram        = __section_begin(".data");



122    data_rom        = __section_begin(".data_init");



123    data_rom_end    = __section_end(".data_init");



124



125    /* CODE RAM */



126    #pragma section = "__CODE_ROM"



127    #pragma section = "__CODE_RAM"



128    code_ram        = __section_begin("__CODE_RAM");



129    code_rom        = __section_begin("__CODE_ROM");



130    code_rom_end    = __section_end("__CODE_ROM");



131



132    /* BSS */



133    bss_start       = __section_begin(".bss");



134    bss_end         = __section_end(".bss");



135



136    custom_ram      = __section_begin(".customSection");



137    custom_rom      = __section_begin(".customSection_init");



138    custom_rom_end  = __section_end(".customSection_init");



139    



140#elif defined (__ARMCC_VERSION)



141    /* VECTOR TABLE*/



142    uint8_t * vector_table_size = (uint8_t *)__RAM_VECTOR_TABLE_SIZE;



143    uint32_t * vector_rom    = (uint32_t *)__VECTOR_ROM;



144    uint32_t * vector_ram    = (uint32_t *)__VECTOR_RAM;



145#else



146    extern uint32_t __DATA_ROM[];



147    extern uint32_t __DATA_RAM[];



148    extern uint32_t __DATA_END[];



149



150    extern uint32_t __CODE_RAM[];



151    extern uint32_t __CODE_ROM[];



152    extern uint32_t __CODE_END[];



153



154    extern uint32_t __BSS_START[];



155    extern uint32_t __BSS_END[];



156



157    extern uint32_t __CUSTOM_ROM[];



158    extern uint32_t __CUSTOM_END[];



159



160    /* Data */



161    data_ram        = (uint8_t *)__DATA_RAM;



162    data_rom        = (uint8_t *)__DATA_ROM;



163    data_rom_end    = (uint8_t *)__DATA_END;



164    /* CODE RAM */



165    code_ram        = (uint8_t *)__CODE_RAM;



166    code_rom        = (uint8_t *)__CODE_ROM;



167    code_rom_end    = (uint8_t *)__CODE_END;



168    /* BSS */



169    bss_start       = (uint8_t *)__BSS_START;



170    bss_end         = (uint8_t *)__BSS_END;



171



172	/* Custom section */



173    custom_ram      = CUSTOMSECTION_SECTION_START;



174    custom_rom      = (uint8_t *)__CUSTOM_ROM;



175    custom_rom_end  = (uint8_t *)__CUSTOM_END;



176



177#endif



178



179#if !defined(__ARMCC_VERSION)



180    /* Copy initialized data from ROM to RAM */



181    while (data_rom_end != data_rom)



182    {



183        *data_ram = *data_rom;



184        data_ram++;



185        data_rom++;



186    }



187



188    /* Copy functions from ROM to RAM */



189    while (code_rom_end != code_rom)



190    {



191        *code_ram = *code_rom;



192        code_ram++;



193        code_rom++;



194    }



195



196    /* Clear the zero-initialized data section */



197    while(bss_end != bss_start)



198    {



199        *bss_start = 0;



200        bss_start++;



201    }



202



203    /* Copy customsection rom to ram */



204    while(custom_rom_end != custom_rom)



205    {



206        *custom_ram = *custom_rom;



207        custom_rom++;



208        custom_ram++;



209    }



210#endif



211    coreId = (uint8_t)GET_CORE_ID();



212#if defined (__ARMCC_VERSION)



213        /* Copy the vector table from ROM to RAM */



214                /* Workaround */



215        for (n = 0; n < (((uint32_t)(vector_table_size))/sizeof(uint32_t)); n++)



216        {



217            vector_ram[n] = vector_rom[n];



218        }



219        /* Point the VTOR to the position of vector table */



220         *s_vectors[coreId] = (uint32_t) __VECTOR_RAM;



221#else



222    /* Check if VECTOR_TABLE copy is needed */



223    if (__VECTOR_RAM != __VECTOR_TABLE)



224    {



225        /* Copy the vector table from ROM to RAM */



226        for (n = 0; n < (((uint32_t)__RAM_VECTOR_TABLE_SIZE)/sizeof(uint32_t)); n++)



227        {



228            __VECTOR_RAM[n] = __VECTOR_TABLE[n];



229        }



230        /* Point the VTOR to the position of vector table */



231        *s_vectors[coreId] = (uint32_t)__VECTOR_RAM;



232    }



233    else



234    {



235        /* Point the VTOR to the position of vector table */



236        *s_vectors[coreId] = (uint32_t)__VECTOR_TABLE;



237    }



238#endif



239



240}
