121/*FUNCTION**********************************************************************



122 *



123 * Function Name : LIN_LPUART_DRV_Init



124 * Description   : This function initializes a LPUART instance for operation.



125 * This function will initialize the run-time state structure to keep track of



126 * the on-going transfers, initialize the module to user defined settings and



127 * default settings, set break field length to be 13 bit times minimum, enable



128 * the break detect interrupt, Rx complete interrupt, frame error detect interrupt,



129 * and enable the LPUART module transmitter and receiver.



130 *



131 * Implements    : LIN_LPUART_DRV_Init_Activity



132 *END**************************************************************************/



133status_t LIN_LPUART_DRV_Init(uint32_t instance,



134                             lin_user_config_t * linUserConfig,



135                             lin_state_t * linCurrentState)



136{



137    /* Assert parameters. */



138    DEV_ASSERT(linUserConfig && linCurrentState);



139    DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



140



141    uint32_t linSourceClockFreq = 0U;



142    status_t retVal = STATUS_SUCCESS;



143



144    /* Get base address of the LPUART instance. */



145    LPUART_Type * base = g_linLpuartBase[instance];



146



147    /* Get the LPUART clock as configured in the clock manager */



148    (void)CLOCK_SYS_GetFreq(s_linLpuartClkName[instance], &linSourceClockFreq);



149



150    /* Check if current instance is clock gated off. */



151    DEV_ASSERT(linSourceClockFreq > 0U);



152



153    /* Check if current instance is already initialized. */



154    DEV_ASSERT(g_linStatePtr[instance] == NULL);



155



156    /* Save runtime structure pointer. */



157    g_linStatePtr[instance] = linCurrentState;



158



159    /* Save LIN user config structure pointer. */



160    g_linUserconfigPtr[instance] = linUserConfig;



161



162    /* Clear linSourceClockFreq value */



163    linCurrentState->linSourceClockFreq = linSourceClockFreq;



164



165    /* Init LPUART */



166    LPUART_Init(base);



167



168    /* Create txCompleted and rxCompleted semaphores as counting, with init value is 0 */



169    if (OSIF_SemaCreate(&linCurrentState->txCompleted, 0) != STATUS_SUCCESS)



170    {



171        retVal = STATUS_ERROR;



172    }



173    else if (OSIF_SemaCreate(&linCurrentState->rxCompleted, 0) != STATUS_SUCCESS)



174    {



175        retVal = STATUS_ERROR;



176    }



177    else



178    {



179        /* if autobaud is enabled */



180        if ((linUserConfig->autobaudEnable) && (linUserConfig->nodeFunction == (bool)SLAVE))



181        {



182            /* Setting Slave's baudrate to 19200 will help Slave node */



183            /* always detect LIN Break from Master */



184            linUserConfig->baudRate = 19200U;



185            linCurrentState->fallingEdgeInterruptCount = 0U;



186            linCurrentState->baudrateEvalEnable = true;



187            s_previousTwoBitTimeLength[instance] = 0U;



188            s_countMeasure[instance] = 0U;



189            s_timeMeasure[instance] = 0U;



190        }



191



192        /* Set baud rate to User's value */



193        (void)LPUART_DRV_SetBaudRate(instance, linUserConfig->baudRate);



194



195        /* Set 8 bit counts per char */



196        LPUART_SetBitCountPerChar(base, LPUART_8_BITS_PER_CHAR, false);



197



198        /* Set no parity mode */



199        LPUART_SetParityMode(base, LPUART_PARITY_DISABLED);



200



201        /* One stop bit */



202        LPUART_SetStopBitCount(base, LPUART_ONE_STOP_BIT);



203



204        /* Check if the current node is MASTER */



205        if (linUserConfig->nodeFunction == (bool)MASTER)



206        {



207            /* Set Break char length as 13 bits minimum */



208            LPUART_SetBreakCharTransmitLength(base, LPUART_BREAK_CHAR_13_BIT_MINIMUM);



209        }



210



211        /* Set Break char detect length as 13 bits minimum */



212        LPUART_SetBreakCharDetectLength(base, LPUART_BREAK_CHAR_13_BIT_MINIMUM);



213



214        /* Enable RX complete interrupt */



215        LPUART_SetIntMode(base, LPUART_INT_RX_DATA_REG_FULL, true);



216



217        /* Enable frame error interrupt */



218        LPUART_SetIntMode(base, LPUART_INT_FRAME_ERR_FLAG, true);



219



220        /* Enable LIN break detect interrupt */



221        LPUART_SetIntMode(base, LPUART_INT_LIN_BREAK_DETECT, true);



222



223        /* Install LIN_RxTx ISR for LPUART instance */



224        INT_SYS_InstallHandler(g_linLpuartRxTxIrqId[instance], g_linLpuartIsrs[instance], (isr_t *)0);



225        /* Enable LPUART interrupts. */



226        INT_SYS_EnableIRQ(g_linLpuartRxTxIrqId[instance]);



227#ifdef LPUART_ERR_IRQS



228        /* Install LIN_ERR ISR for LPUART instance */



229        INT_SYS_InstallHandler(g_linLpuartErrIrqId[instance], g_linLpuartErrIsrs[instance], (isr_t *)0);



230        /* Enable LPUART error interrupts */



231        INT_SYS_EnableIRQ(g_linLpuartErrIrqId[instance]);



232#endif



233        /* Change node's current state to IDLE */



234        linCurrentState->currentNodeState = LIN_NODE_STATE_IDLE;



235



236        /* Clear flags in current LIN state structure */



237        linCurrentState->isTxBusy = false;



238        linCurrentState->isRxBusy = false;



239        linCurrentState->isBusBusy = false;



240        linCurrentState->isRxBlocking = false;



241        linCurrentState->isTxBlocking = false;



242        linCurrentState->timeoutCounterFlag = false;



243        linCurrentState->timeoutCounter = 0U;



244



245        /* Assign wakeup signal to satisfy LIN Specifications specifies that



246         * wakeup signal shall be in range from 250us to 5 ms.



247         */



248        if (linUserConfig->baudRate > 10000U)



249        {



250            /* Wakeup signal will be range from 400us to 800us depend on baudrate */



251            s_wakeupSignal[instance] = 0x80U;



252        }



253        else



254        {



255            /* Wakeup signal will be range from 400us to 4ms depend on baudrate */



256            s_wakeupSignal[instance] = 0xF8U;



257        }



258



259        if (!((linUserConfig->autobaudEnable) && (linUserConfig->nodeFunction == (bool)SLAVE)))



260        {



261            /* Enable the LPUART transmitter and receiver */



262            LPUART_SetTransmitterCmd(base, true);



263            LPUART_SetReceiverCmd(base, true);



264        }



265    }



266



267    return retVal;



268}

270/*FUNCTION**********************************************************************



271 *



272 * Function Name : LIN_LPUART_DRV_Deinit



273 * Description   : This function shuts down the LPUART by disabling interrupts and



274 *                 transmitter/receiver.



275 *



276 * Implements    : LIN_LPUART_DRV_Deinit_Activity



277 *END**************************************************************************/



278void LIN_LPUART_DRV_Deinit(uint32_t instance)



279{



280    /* Assert parameters. */



281    DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



282



283    /* Get base address of the LPUART instance. */



284    LPUART_Type * base = g_linLpuartBase[instance];



285



286    /* Get the current LIN state of this LPUART instance. */



287    lin_state_t * linCurrentState = g_linStatePtr[instance];



288



289    /* Check if current instance is already de-initialized or is gated.*/



290    DEV_ASSERT(linCurrentState != NULL);



291    DEV_ASSERT(linCurrentState->linSourceClockFreq > 0U);



292



293    /* Wait until the data is completely shifted out of shift register */



294    while (!LPUART_GetStatusFlag(base, LPUART_TX_COMPLETE))



295    {



296        /* Do nothing */



297    }



298



299    /* Disable the LPUART transmitter and receiver */



300    LPUART_SetTransmitterCmd(base, false);



301    LPUART_SetReceiverCmd(base, false);



302



303    /* Destroy TX and RX semaphores. */



304    (void)OSIF_SemaDestroy(&linCurrentState->txCompleted);



305    (void)OSIF_SemaDestroy(&linCurrentState->rxCompleted);



306



307    /* Disable LPUART interrupts. */



308    INT_SYS_DisableIRQ(g_linLpuartRxTxIrqId[instance]);



309#ifdef LPUART_ERR_IRQS



310    INT_SYS_DisableIRQ(g_linLpuartErrIrqId[instance]);



311#endif



312



313    /* Change node's current state to UNINIT */



314    linCurrentState->currentNodeState = LIN_NODE_STATE_UNINIT;



315



316    /* Clear our saved pointer to the LIN state structure */



317    g_linStatePtr[instance] = NULL;



318



319}

374/*FUNCTION**********************************************************************



375 *



376 * Function Name : LIN_LPUART_DRV_MakeChecksumByte



377 * Description   : This function calculate checksum for a frame. This function



378 * will return classic or enhanced checksum base on data in



379 * g_linUserconfigPtr[instance] and input parameter.



380 *



381 * Implements    : LIN_LPUART_DRV_SendFrameDataBlocking_Activity



382 *END**************************************************************************/



383static uint8_t LIN_LPUART_DRV_MakeChecksumByte(uint32_t instance,



384                                               const uint8_t * buffer,



385                                               uint8_t sizeBuffer,



386                                               uint8_t PID)



387{



388    /* Get list of PIDs use classic checksum. */



389    const uint8_t *classicPID = g_linUserconfigPtr[instance]->classicPID;



390    const uint8_t numOfClassicPID = g_linUserconfigPtr[instance]->numOfClassicPID;



391    uint8_t checkSum = PID;



392    uint8_t retVal = 0U;



393



394    if(numOfClassicPID == 255U)



395    {



396        /*all frame use enhanced checksum */



397        checkSum = 0U;



398    }



399    else



400    {



401        if(classicPID != NULL)



402        {



403            for (retVal = 0U; retVal < numOfClassicPID; retVal++)



404            {



405                if(checkSum == classicPID[retVal])



406                {



407                    checkSum = 0U;



408                    break;



409                }



410            }



411        }



412    }



413    retVal = LIN_DRV_MakeChecksumByte(buffer, sizeBuffer, checkSum);



414    return retVal;



415}

417/*FUNCTION**********************************************************************



418 *



419 * Function Name : LIN_LPUART_DRV_SendFrameDataBlocking



420 * Description   : This function sends data out through the LPUART module using



421 * blocking method. This function checks if txSize is in range from 1 to 8.



422 * If not, it will return STATUS_ERROR. This function also returns STATUS_ERROR



423 * if node's current state is in SLEEP mode. This function checks if the



424 * isBusBusy is false, if not it will return STATUS_BUSY. The function does not



425 * return until the transmission is complete. If the transmission is successful,



426 * it will return STATUS_SUCCESS. If not, it will return STATUS_TIMEOUT.



427 *



428 * Implements    : LIN_LPUART_DRV_SendFrameDataBlocking_Activity



429 *END**************************************************************************/



430status_t LIN_LPUART_DRV_SendFrameDataBlocking(uint32_t instance,



431                                              const uint8_t * txBuff,



432                                              uint8_t txSize,



433                                              uint32_t timeoutMSec)



434{



435    /* Assert parameters. */



436    DEV_ASSERT(txBuff != NULL);



437    DEV_ASSERT(instance < LPUART_INSTANCE_COUNT);



438



439    /* Get base address of the LPUART instance. */



440    LPUART_Type * base = g_linLpuartBase[instance];



441



442    /* Get the current LIN state of this LPUART instance. */



443    lin_state_t * linCurrentState = g_linStatePtr[instance];



444



445    status_t retVal = STATUS_SUCCESS;



446



447    /* Check whether current mode is sleep mode */



448    bool checkSleepMode = (LIN_NODE_STATE_SLEEP_MODE == linCurrentState->currentNodeState);



449



450    /* Check if txSize > 8 or equal to 0 or node's current state



451     * is in SLEEP mode then return STATUS_ERROR */



452    if ((8U < txSize) || (0U == txSize) || checkSleepMode)



453    {



454        retVal = STATUS_ERROR;



455    }



456    else



457    {



458        /* Check if the LIN Bus is busy */



459        if (linCurrentState->isBusBusy)



460        {



461            retVal = STATUS_BUSY;



462        }



463        else



464        {



465            /* Make the checksum byte. */



466            linCurrentState->checkSum = LIN_LPUART_DRV_MakeChecksumByte(instance, txBuff, txSize, linCurrentState->currentPid);



467



468            /* Update the LIN state structure. */



469            linCurrentState->txBuff = txBuff;



470            /* Add a place for checksum byte */



471            linCurrentState->txSize = (uint8_t)(txSize + 1U);



472            linCurrentState->cntByte = 0U;



473            linCurrentState->currentEventId = LIN_NO_EVENT;



474            linCurrentState->isBusBusy = true;



475            linCurrentState->isTxBusy = true;



476            linCurrentState->isTxBlocking = true;



477



478            /* Set Break char detect length as 10 bits minimum */



479            LPUART_SetBreakCharDetectLength(base, LPUART_BREAK_CHAR_10_BIT_MINIMUM);



480



481            /* Set node's current state to SEND_DATA */



482            linCurrentState->currentNodeState = LIN_NODE_STATE_SEND_DATA;



483



484            /* Start sending data */



485            LPUART_Putchar(base, *linCurrentState->txBuff);



486



487            /* Wait until the transmission is complete. */



488            retVal = LIN_LPUART_DRV_WaitComplete(instance, timeoutMSec);



489



490            /* Clear Tx Blocking flag */



491            linCurrentState->isTxBlocking = false;



492



493            /* Clear Tx Busy flag. */



494            linCurrentState->isTxBusy = false;



495



496            /* Change node's current state to IDLE */



497            (void)LIN_LPUART_DRV_GotoIdleState(instance);



498        }



499    }



500



501    return retVal;



502}

574/*FUNCTION**********************************************************************

579 * function to ascertain the state of the current transmission:



580 * in progress (or busy that is STATUS_BUSY) or timeout (if timeout has occurred that is



581 * STATUS_TIMEOUT) or complete (success that is STATUS_SUCCESS).



582 * In addition, if the transmission is still in progress, the user can obtain the number



583 * of bytes that still needed to transmit.



584 *



585 * Implements    : LIN_LPUART_DRV_GetTransmitStatus_Activity

605        {



606            retVal = STATUS_BUSY;



607        }



608        else



609        {



610            retVal = STATUS_TIMEOUT;



611        }



612    }



613



614    return retVal;

636{
