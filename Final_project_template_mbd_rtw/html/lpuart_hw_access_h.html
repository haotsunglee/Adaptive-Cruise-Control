201/*!



202 * @brief Enable/Disable the LPUART transmitter.



203 *



204 * This function enables or disables the LPUART transmitter, based on the



205 * parameter received.



206 *



207 *



208 * @param base LPUART base pointer.



209 * @param enable Enable(true) or disable(false) transmitter.



210 */



211static inline void LPUART_SetTransmitterCmd(LPUART_Type * base, bool enable)



212{



213    base->CTRL = (base->CTRL & ~LPUART_CTRL_TE_MASK) | ((enable ? 1UL : 0UL) << LPUART_CTRL_TE_SHIFT);



214    /* Wait for the register write operation to complete */



215    while((bool)((base->CTRL & LPUART_CTRL_TE_MASK) != 0U) != enable) {}



216}

222 * parameter received.



223 *



224 *

226 * @param enable Enable(true) or disable(false) receiver.



227 */



228static inline void LPUART_SetReceiverCmd(LPUART_Type * base, bool enable)



229{



230    base->CTRL = (base->CTRL & ~LPUART_CTRL_RE_MASK) | ((enable ? 1UL : 0UL) << LPUART_CTRL_RE_SHIFT);



231    /* Wait for the register write operation to complete */



232    while((bool)((base->CTRL & LPUART_CTRL_RE_MASK) != 0U) != enable) {}



233}

235/*!



236 * @brief Sets the LPUART baud rate modulo divisor.



237 *



238 * This function sets the LPUART baud rate modulo divisor.



239 *



240 *



241 * @param base LPUART base pointer.



242 * @param baudRateDivisor The baud rate modulo division "SBR"



243 */



244static inline void LPUART_SetBaudRateDivisor(LPUART_Type * base, uint32_t baudRateDivisor)



245{



246    DEV_ASSERT((baudRateDivisor <= 0x1FFFU) && (baudRateDivisor >= 1U));



247    uint32_t baudRegValTemp;



248



249    baudRegValTemp = base->BAUD;



250    baudRegValTemp &= ~(LPUART_BAUD_SBR_MASK);



251    /* Removed the shift operation as the SBR field position is zero; shifting with 0 violates MISRA */



252    baudRegValTemp |= baudRateDivisor & LPUART_BAUD_SBR_MASK;



253    base->BAUD = baudRegValTemp;



254}



255



256/*!



257 * @brief Gets the LPUART baud rate modulo divisor.



258 *



259 * This function gets the LPUART baud rate modulo divisor.



260 *



261 *



262 * @param base LPUART base pointer.



263 * @return The baud rate modulo division "SBR"



264 */



265static inline uint16_t LPUART_GetBaudRateDivisor(const LPUART_Type * base)



266{



267    return ((uint16_t)((base->BAUD & LPUART_BAUD_SBR_MASK) >> LPUART_BAUD_SBR_SHIFT));



268}

270#if FEATURE_LPUART_HAS_BAUD_RATE_OVER_SAMPLING_SUPPORT



271/*!



272 * @brief Sets the LPUART baud rate oversampling ratio



273 *



274 * This function sets the LPUART baud rate oversampling ratio.



275 * (Note: Feature available on select LPUART instances used together with baud rate programming)



276 * The oversampling ratio should be set between 4x (00011) and 32x (11111). Writing



277 * an invalid oversampling ratio results in an error and is set to a default



278 * 16x (01111) oversampling ratio.



279 * Disable the transmitter/receiver before calling this function.



280 *



281 *



282 * @param base LPUART base pointer.



283 * @param overSamplingRatio The oversampling ratio "OSR"



284 */



285static inline void LPUART_SetOversamplingRatio(LPUART_Type * base, uint32_t overSamplingRatio)



286{



287    DEV_ASSERT(overSamplingRatio <= 0x1FU);



288    uint32_t baudRegValTemp;



289



290    baudRegValTemp = base->BAUD;



291    baudRegValTemp &= ~(LPUART_BAUD_OSR_MASK);



292    baudRegValTemp |= LPUART_BAUD_OSR(overSamplingRatio);



293    base->BAUD = baudRegValTemp;



294}
